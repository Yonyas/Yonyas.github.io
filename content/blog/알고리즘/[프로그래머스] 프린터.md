function solution(priorities, location) {
  let cnt = 0
  let copy = [...priorities]
  let copySorted = copy.sort().reverse()
  while (JSON.stringify(priorities) !== JSON.stringify(copySorted)) {
    priorities.push(priorities.shift())
    cnt++
  }
  let idx = location - cnt
  console.log('cnt', cnt)
  console.log('idx', idx)
  return idx < 0 ? priorities.length + 1 + idx : idx + 1
}

solution(priorities, 2)
