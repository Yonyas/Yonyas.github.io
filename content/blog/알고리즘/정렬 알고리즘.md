## 삽입정렬

인덱스 1번째부터 오른쪽으로 훑는다.
훑는 부분의 왼쪽 배열들과 비교해서 왼쪽숫자보다 크면 오른쪽, 오른쪽 숫자보다 작으면 왼쪽에 넣는 식으로 정렬해나간다.

```js
//삽입정렬 구현

let arr = [1, 5, 2, 7, 3, 1]

function insertOrder(arr) {
  for (let i = 1; i < arr.length; i++) {
    for (let j = 0; j < i; j++) {
      if (arr[j] <= arr[i] && arr[i] < arr[j + 1]) {
        let willChangedNum = arr[i]
        arr.splice(i, 1)
        arr.splice(j + 1, 0, willChangedNum)
      }
    }
  }
}

insertOrder(arr)
```
